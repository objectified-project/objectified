Replay protection

Reject events with timestamps older than 5 minutes. Rotate immediately if the secret is exposed — both old and new secrets validate during the 24h grace period.

Verification sample

import { createHmac, timingSafeEqual } from 'crypto';

export function verify(req, secret) {
  const header = req.headers['x-objectified-signature'];
  const [t, v1] = header.split(',').map(p => p.split('=')[1]);
  const body  = req.rawBody.toString();
  const hmac  = createHmac('sha256', secret)
                  .update(t + '.' + body)
                  .digest('hex');
  const ok    = timingSafeEqual(Buffer.from(v1), Buffer.from(hmac));
  const fresh = Math.abs(Date.now()/1000 - +t) < 300;
  return ok && fresh;
}
