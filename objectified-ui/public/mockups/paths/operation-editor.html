# PATCH a customer with optimistic concurrency

  -H 'Authorization: Bearer $TOKEN' \
  -H 'X-Tenant-Id: acme-prod' \
  -H 'If-Match: W/"v17"' \
  -H 'Idempotency-Key: f1c4-7e21-...' \
  -H 'Content-Type: application/merge-patch+json' \
  -d '{"name":"Acme Corp"}'
