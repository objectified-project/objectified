Programmatic export

Same job pipeline via the public REST API

# Kick off an export
curl -X POST https://api.objectified.dev/api/v1/analytics/export \
  -H "Authorization: Bearer $OBJ_TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "report_type": "executive_dashboard",
    "format": "pdf",
    "from": "2026-01-22",
    "to": "2026-04-22",
    "filters": { "project": "payments-api" }
  }'

# → 202 Accepted
{ "job_id": "job_3f2a91", "status": "queued" }

# Poll until ready, then GET the signed download_url
curl https://api.objectified.dev/api/v1/analytics/export/job_3f2a91 \
  -H "Authorization: Bearer $OBJ_TOKEN"
