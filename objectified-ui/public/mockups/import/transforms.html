Stage 4 · enrich-tags.js

Sandboxed V8 · 64 MB / 5 s budget · no network access

JS 
// signature: (schema, ctx) => schema
export default function enrichTags(schema, ctx) {
  if (!schema.properties) return schema;

  for (const [name, prop] of Object.entries(schema.properties)) {
    // auto-tag PII fields
    if (/email|phone|ssn|name/i.test(name)) {
      prop["x-pii"] = true;
      prop["x-tags"] = [...(prop["x-tags"] ?? []), "pii"];
    }
    // auto-tag monetary fields
    if (/amount|price|fee|total/i.test(name) && prop.type === "integer") {
      prop["x-monetary"] = true;
      prop["x-currency-source"] = "sibling:currency";
    }
  }
  return schema;
}
