# preset: acme-default v3.1.0
id: preset_acme_default
version: "3.1.0"
source:
  allow: [openapi-3.1, swagger-2]
  migrate_swagger: true
target:
  project: acme-platform
  version: "{{ git.branch }}"
naming:
  style: snake_case
  enum:  UPPER_CASE
layout:
  algorithm: auto
approvals:
  required: 1
  reviewers: [@platform-eng]
auto_actions:
  - on: field_added
    do: tag:auto-added
  - on: enum_value_added
    do: notify:slack#schemas
on_warning: save_draft
on_failure: rollback
