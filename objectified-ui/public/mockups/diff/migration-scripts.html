import { z } from "zod";
import type { UserV2, UserV3 } from "./types";

export const userV3Schema = z.object({
  name: z.string(),
  email: z.string().email(),
  dateOfBirth: z.string().date().optional(),
  contacts: z.array(z.object({
    type: z.enum(["phone", "email", "address"]),
    value: z.string(),
  })),
  locale: z.string().optional(),
});

export function migrateUser(v2: UserV2): UserV3 {
  // REMOVED → DERIVED  ·  age has no inverse, dob is preferred when available.
  const dateOfBirth = v2.dateOfBirth
    ?? deriveDobFromAge(v2.age, v2.createdAt);

  // FLATTENED → STRUCTURED  ·  preserve original phone as a single contact.
  const contacts = v2.phone
    ? [{ type: "phone" as const, value: v2.phone }]
    : [];

  const v3: UserV3 = {
    name: v2.name,
    email: v2.email,
    dateOfBirth,
    contacts,
    locale: v2.locale ?? "en-US",
  };

  return userV3Schema.parse(v3);
}

function deriveDobFromAge(age?: number, ref?: string) {
  if (age == null) return undefined;
  const base = ref ? new Date(ref) : new Date();
  base.setFullYear(base.getFullYear() - age);
  return base.toISOString().slice(0, 10);
}
