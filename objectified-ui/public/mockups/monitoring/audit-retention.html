Immutability proof · PostgreSQL role grant

enforced at db role level 
-- Role: app_user (the application connects as this role)
GRANT INSERT ON audit_log TO app_user;

-- Explicit revokes — application physically cannot mutate
REVOKE UPDATE,
       DELETE,
       TRUNCATE
  ON audit_log FROM app_user;

-- Replication for archival uses a separate read-only role
GRANT SELECT ON audit_log TO archive_reader;

-- Partition pruning DDL is restricted to a privileged migration role
ALTER TABLE audit_log
  ADD CONSTRAINT audit_log_no_future
  CHECK (occurred_at <= now() + interval '5 minutes');
