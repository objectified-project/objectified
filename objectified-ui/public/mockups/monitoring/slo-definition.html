slo_definition

data model · PostgreSQL

CREATE TABLE slo_definition (
  id            UUID        PRIMARY KEY,
  name          VARCHAR(120) NOT NULL,
  service       VARCHAR(64)  NOT NULL,
  metric_type   slo_metric_t NOT NULL,    -- enum
  target        NUMERIC(8,4) NOT NULL,
  target_unit   slo_unit_t   NOT NULL,
  window_days   INTEGER      NOT NULL DEFAULT 30,
  source_query  TEXT         NOT NULL,
  created_by    UUID REFERENCES users(id),
  created_at    TIMESTAMPTZ DEFAULT now(),
  CHECK (window_days BETWEEN 1 AND 90)
);

CREATE TYPE slo_metric_t AS ENUM
  ('availability','latency','error_rate');
CREATE TYPE slo_unit_t   AS ENUM
  ('percent','ms');
