Each Review triggers an independent User fetch. Suggested DataLoader batching code:

const userLoader = new DataLoader(async (ids) => {
  const users = await fetchUsersByIds(ids);
  return ids.map(id => users.find(u => u.id === id));
});

const resolvers = {
  Review: {
    author: (review) => userLoader.load(review.authorId),
  },
};
