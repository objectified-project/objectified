Verify the signature on your end

Compute HMAC-SHA256(secret, timestamp + "." + raw_body) and compare to the X-Objectified-Signature header (constant-time).

// Node.js
const crypto = require('crypto');
const sig = req.headers['x-objectified-signature'];
const ts  = req.headers['x-objectified-timestamp'];
const expected = 'sha256=' +
  crypto.createHmac('sha256', secret)
        .update(`${ts}.${rawBody}`).digest('hex');
if (!crypto.timingSafeEqual(Buffer.from(sig), Buffer.from(expected))) throw new Error('invalid signature');
