-- Filter: active customers with valid email
-- Node: filter_active_customers  |  Pipeline v2.4

SELECT
  c.customer_id,
  c.email,
  c.first_name,
  c.last_name,
  c.status,
  c.created_at,
  c.plan_id,
  COALESCE(c.segment, 'unassigned') AS segment,
  e.event_count,
  e.last_event_at
FROM source_postgres_customers c
LEFT JOIN source_s3_events e
  ON c.customer_id = e.customer_id
WHERE
  c.status = 'active'
  AND c.created_at > TIMESTAMP '2024-01-01 00:00:00'
  AND c.email IS NOT NULL
  AND c.email NOT LIKE '%+test%'
