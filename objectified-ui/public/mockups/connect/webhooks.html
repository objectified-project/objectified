Verify on your endpoint

Reject any request whose computed signature does not match the header.

Node.js
import crypto from 'crypto';
const sig = req.headers['x-objectified-signature'];
const want = crypto.createHmac('sha256', SECRET)
  .update(req.rawBody).digest('hex');
if (sig !== want) return res.status(401).end();
Python
import hmac, hashlib
sig = request.headers["X-Objectified-Signature"]
want = hmac.new(SECRET, request.body,
                hashlib.sha256).hexdigest()
if not hmac.compare_digest(sig, want):
    abort(401)
Go
mac := hmac.New(sha256.New, []byte(SECRET))
mac.Write(body)
want := hex.EncodeToString(mac.Sum(nil))
if !hmac.Equal([]byte(sig), []byte(want)) {
    http.Error(w, "bad signature", 401)
}
