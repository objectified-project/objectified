Community Connector SDK

Build & publish your own connector — @objectified/connector-sdk v1.4.0

Quickstart · 4 commands
1Init

Scaffold project skeleton with manifest, Dockerfile, tests.

$ connector init mailgun
2Test

Run the harness against your implementation locally.

$ connector test --watch
3Validate

Manifest schema check + Docker build + interface compliance.

$ connector validate
4Publish

Submit to the marketplace pending review.

$ connector publish

 3/** Mailgun email events & mailing lists. */
 4export default class MailgunConnector extends ConnectorBase {
 5  async authenticate(creds: { apiKey: string; domain: string }) {
 6    this.client = mailgun({ apiKey: creds.apiKey, domain: creds.domain });
 7  }
 8
 9  async testConnection(): Promise<void> {
10    await this.client.domains.get(this.config.domain);
11  }
12
13  async discover(): Promise<DiscoverResult> {
14    return {
15      objects: [
16        { name: 'event', fields: EVENT_SCHEMA },
17        { name: 'mailing_list', fields: LIST_SCHEMA },
18        { name: 'list_member', fields: MEMBER_SCHEMA, parent: 'mailing_list' },
19      ],
20    };
21  }
22
23  async read(object: string, cursor?: string): Promise<RecordBatch> {
