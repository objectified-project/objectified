name: Publish schema package
on:
  push:
    tags: ['v*.*.*']
  pull_request:
    paths: ['schemas/**', 'obj-package.json']

jobs:
  publish:
    runs-on: ubuntu-latest
    permissions:
      contents: read
      id-token: write   # for OIDC trust
    steps:
      - uses: actions/checkout@v4

      - uses: objectified/publish-action@v1
        with:
          token: ${{ secrets.OBJ_PUBLISH_TOKEN }}
          access: public
          dist-tag: ${{ contains(github.ref, '-') && 'next' || 'latest' }}
          dry-run: ${{ github.event_name == 'pull_request' }}
          verify-signatures: true
          comment-on-release: true
